---
title: "Get Campaign Leads"
api: "GET https://server.smartlead.ai/api/v1/campaigns/{campaign_id}/leads"
description: "Retrieve paginated list of leads in a campaign with advanced filtering"
---

<Note>
Supports filtering by status, category, email engagement, and date. Returns lead details with custom fields. Max 100 per page.
</Note>

## Path Parameters

<ParamField path="campaign_id" type="number" required>
  The campaign ID
</ParamField>

## Query Parameters

<ParamField query="api_key" type="string" required>
  Your SmartLead API key
</ParamField>

<ParamField query="offset" type="number" default="0">
  Pagination offset
</ParamField>

<ParamField query="limit" type="number" default="100" max="100">
  Number of leads to return
</ParamField>

<ParamField query="status" type="string">
  Filter by status: STARTED, INPROGRESS, COMPLETED, PAUSED, STOPPED
</ParamField>

<ParamField query="lead_category_id" type="number">
  Filter by category ID
</ParamField>

<ParamField query="emailStatus" type="string">
  Filter by email status: is_opened, is_clicked, is_replied, is_bounced, is_unsubscribed
</ParamField>

<RequestExample>

```bash cURL
curl "https://server.smartlead.ai/api/v1/campaigns/123/leads?api_key=YOUR_KEY&limit=50&status=INPROGRESS"
```

```python Python
import requests

API_KEY = "YOUR_API_KEY"
campaign_id = 123

response = requests.get(
    f"https://server.smartlead.ai/api/v1/campaigns/{campaign_id}/leads",
    params={
        "api_key": API_KEY,
        "limit": 50,
        "status": "INPROGRESS"
    }
)

result = response.json()
print(f"Total leads: {result.get('total_leads', 0)}")
```

```javascript JavaScript
const API_KEY = 'YOUR_API_KEY';
const campaignId = 123;

const response = await fetch(
  `https://server.smartlead.ai/api/v1/campaigns/${campaignId}/leads?api_key=${API_KEY}&limit=50&status=INPROGRESS`
);

const result = await response.json();
console.log(`Total leads: ${result.total_leads}`);
```

</RequestExample>
